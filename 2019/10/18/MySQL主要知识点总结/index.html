MySQL主要知识点总结

#这是MySQL主要知识点汇总和部分例子
1、创建语句
1.创建数据库
create database 库名;
2.创建表
create table 表名(
字段名1 数据类型 约束，
字段名2 数据类型 约束，
字段名3 数据类型 约束
//约束可有可无，最后一个字段不加逗号
//最常用的数据类型
//int
//char 字符串类型，固定长度
//varchar 字符串长度，可变长度
//float
)
2、插入语句
insert into 表名（字段1，字段2）
values（数据1，数据2）
3、查询语句
select 字段1，字段2，字段3
form 表名
4、删除语句
删除库
drop database 库名;
删除表
drop table 表名;
删除表中所有的数据
delete from 表名;
删除表中指定的数据
delete from 表名 where …
5、更改语句
update 表名 set
#更改表名
ALTER TABLE 旧表名 RENAME[TO]新表名
#修改数据类型
ALTER TABLE 表名 MODIFY 列名的数据类型
#修改列名
ALTER TABLE <表名> CHANGE 旧列名 新列名 新数据类型
删除列
ALTER TABLE 表名 drop 列名
添加列
ALTER TABLE 表名 add 新列名的数据类型 [约束] [first | after 已存在列名]

#约束
1.主键约束(primary key)
主键：一行的唯一标识
表中的其它数据依赖于主键
不允许重复非空
2.非空约束(not null)
3.唯一约束(unique)
此列不允许重复
4.默认值约束(default)
5.check约束(check)
6.外键约束(foreign key)
外键：此表中一个字段，引用另外一张表中字段数据，
并且被引用的字段是另一张表的主键。那么此表中的字
段就称为本张表的外键

MySQL数据库的设计规范
数据库设计的三大范式
第一范式：
表中的字段都是不可分割的最小数据单元。
第二范式：
每张表只能描述一件事情；
每张表都要有主键。
第三范式：
非主键字段要依赖于主键字段；
换句话说，非主键字段与主键字段
直接相关。
在设计表时，尽可能依靠三大范式；
但有的时候，为了性能会牺牲范式；

        连接
                                   把多张表连接成一张表,要注意
                                   一般连接有主外键关系的表
           连接分类：
           普通连接  
           内连接
             inner join .. on...    inner join用内连接方式连接两张表
             on                      后边写的是条件
           外连接（左连接与右连接）
               左连接
                                   left join..on...
                                   左边的表显示所有数据，右边的表只显示匹配上的数据
               右连接
                   right join...on...
                                   右边的表显示全部数据左边的表只显示匹配上的
           全连接
                                   左右边都全部显示

   #order by排序
   #把StudentInfo表按年纪从小到大排序
   SELECT * FROM StudentInfo ORDER BY 
       age ASC;
   #把StudentInfo表按年纪从大到小排序
   SELECT * FROM StudentInfo ORDER BY 
       age DESC;
   #group by分组
   #把某张表按照某个字段进行分组
   #查询出每个学生的平均成绩，结果要
   #显示的字段为，学生学号 平均成绩
   SELECT sid,AVG(score) '平均成绩'
   FROM score GROUP BY sid;
   #group by ... havig..筛选出来符合条件
   #的组
   #查询出全部及格学生的平均成绩
   SELECT sid,AVG(score) 'avg' FROM score
   GROUP BY sid HAVING MIN(score)>60;
   #查询出考试门数不够四门的学生学号
   SELECT sid FROM score GROUP BY sid
       HAVING COUNT(*)<4;
   #查询出“计算机基础”的成绩
   SELECT coid,score FROM score WHERE
   coid=(
   SELECT coid FROM course 
                WHERE cnam='计算机基础');
   #查询出每门课的平均成绩
   SELECT coid,AVG(score)'平均成绩' FROM
                  score GROUP BY coid; 
   #查询出‘周三’学生的平均成绩
   SELECT AVG(score) '平均成绩' FROM
                  score WHERE sid=(
   SELECT sid FROM StudentInfo 
                  WHERE snam='周三');
   #连接
   #普通连接
   SELECT * FROM StudentInfo s,score c
                         WHERE 
                         s.sid=c.sid;
   #查询出每个人的平均成绩，要显示的
   #字段为，姓名 平均成绩
   SELECT snam,AVG(score) FROM 
                StudentInfo s,score c
                WHERE 
                s.sid=c.sid
                GROUP BY s.sid;
   #内连接
   SELECT * FROM StudentInfo s 
                 INNER JOIN
                 score c
                 ON
                 s.sid=c.sid;
   SELECT snam,AVG(score) FROM 
                StudentInfo s
                INNER JOIN             
                score c
                ON 
                s.sid=c.sid
                GROUP BY s.sid;

   #左连接
   SELECT * FROM StudentInfo s 
                 LEFT JOIN
                 score c
                 ON
                 s.sid=c.sid;
   #查询出有不及格成绩的学生信息，
   #缺考算不及格
   SELECT * FROM StudentInfo WHERE sid 
   IN(
   SELECT sid FROM score
             GROUP BY sid 
             HAVING MIN(score)<60
   UNION
   SELECT sid FROM score 
                GROUP BY sid
                HAVING 
                COUNT(*)<
                (SELECT COUNT(*) FROM 
                  course));
